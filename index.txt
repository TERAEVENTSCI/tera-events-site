    <script>
        // --- BASE DE DONNÉES LOCALE ---
        let state = {
            config: { target: 30000000 },
            admins: [{ user: "admin@corp.com", pass: "1234" }],
            users: [
                { id: 1, name: "Jean Vendeur", role: "Commercial", rate: 10 },
                { id: 2, name: "Awa Koné", role: "Commercial", rate: 12 }
            ],
            deals: [
                { id: 101, title: "SODEFOR", user: "Jean Vendeur", price: 5000000, stage: 'gagne', date: '2026-03-20' },
                { id: 102, title: "Banque Atlantique", user: "Awa Koné", price: 12000000, stage: 'devis', date: '2026-03-25' }
            ]
        };

        let currentChart = null;

        // --- AUTHENTIFICATION ---
        function handleLogin() {
            const u = document.getElementById('adm-user').value;
            const p = document.getElementById('adm-pass').value;
            if(state.admins.some(a => a.user === u && a.pass === p)) {
                document.getElementById('login-screen').style.display = 'none';
                initApp();
            } else {
                document.getElementById('login-err').style.display = 'block';
            }
        }

        // --- NAVIGATION ---
        function showView(id) {
            document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
            document.querySelectorAll('.nav-links a').forEach(a => a.classList.remove('active'));
            document.getElementById('view-'+id).classList.add('active');
            document.getElementById('l-'+id).classList.add('active');
            updateAll();
        }

        // --- INITIALISATION ---
        function initApp() {
            const sel = document.getElementById('period-filter');
            const months = ["Janvier", "Février", "Mars", "Avril", "Mai", "Juin", "Juillet", "Août", "Septembre", "Octobre", "Novembre", "Décembre"];
            const now = new Date();
            months.forEach((m, i) => {
                let opt = document.createElement('option');
                opt.value = (i+1).toString().padStart(2, '0');
                opt.text = m + " 2026";
                if(i === now.getMonth()) opt.selected = true;
                sel.appendChild(opt);
            });
            updateAll();
        }

        // --- CORE CALCULATIONS & UI UPDATE ---
        function updateAll() {
            const m = document.getElementById('period-filter').value;
            const periodStr = `2026-${m}`;
            const today = new Date().toISOString().split('T')[0];

            // Filtrage précis
            const mDeals = state.deals.filter(d => d.date.startsWith(periodStr) && d.stage === 'gagne');
            const todayCA = state.deals.filter(d => d.date === today && d.stage === 'gagne').reduce((s,d)=>s+d.price, 0);

            // Dashboard Stats
            const totalCA = mDeals.reduce((s,d) => s + d.price, 0);
            const totalCom = mDeals.reduce((s,d) => {
                const u = state.users.find(x => x.name === d.user);
                return s + (d.price * (u ? u.rate : 0) / 100);
            }, 0);

            document.getElementById('dash-ca').innerText = totalCA.toLocaleString() + " FCFA";
            document.getElementById('dash-com').innerText = totalCom.toLocaleString() + " FCFA";
            document.getElementById('dash-today').innerText = todayCA.toLocaleString() + " FCFA";
            document.getElementById('dash-kpi').innerText = Math.round((totalCA / state.config.target)*100) + "%";

            // Tableau Dashboard
            let dashHTML = "";
            state.users.filter(u => u.role === 'Commercial').forEach(u => {
                const uDeals = mDeals.filter(d => d.user === u.name);
                const uCA = uDeals.reduce((s,d) => s + d.price, 0);
                dashHTML += `<tr><td>${u.name}</td><td>${uDeals.length}</td><td>${uCA.toLocaleString()} FCFA</td><td>${(uCA*u.rate/100).toLocaleString()} FCFA</td></tr>`;
            });
            document.getElementById('dash-table-body').innerHTML = dashHTML;

            // Pipeline
            renderKanban();
            // Admin Table
            renderAdminTable();
            // Chart
            if(document.getElementById('view-reports').classList.contains('active')) renderBI();
        }

        // --- GESTION PIPELINE (CRUD) ---
        function renderKanban() {
            const stages = [{id:'prospection', t:'Prospection'}, {id:'devis', t:'Négociation'}, {id:'gagne', t:'Gagné'}];
            document.getElementById('kanban-board').innerHTML = stages.map(s => `
                <div class="column">
                    <h4 style="margin-bottom:15px; display:flex; justify-content:space-between;">
                        ${s.t} <span>${state.deals.filter(d=>d.stage===s.id).length}</span>
                    </h4>
                    ${state.deals.filter(d => d.stage === s.id).map(d => `
                        <div class="deal-card">
                            <div class="deal-actions">
                                <i class="fas fa-edit" onclick="editDeal(${d.id})" style="color:var(--primary)"></i>
                                <i class="fas fa-trash" onclick="deleteDeal(${d.id})" style="color:var(--danger)"></i>
                            </div>
                            <div style="font-weight:700;">${d.title}</div>
                            <div style="font-size:0.75rem; color:gray;">${d.user} | ${d.date}</div>
                            <div style="margin-top:8px; font-weight:700; color:var(--primary);">${d.price.toLocaleString()} FCFA</div>
                            <div style="margin-top:10px;">
                                ${s.id !== 'gagne' ? `<button class="btn btn-ghost" style="padding:4px 8px; font-size:0.6rem;" onclick="moveDeal(${d.id}, 'gagne')">Gagner</button>` : ''}
                            </div>
                        </div>
                    `).join('')}
                </div>
            `).join('');
        }

        function deleteDeal(id) {
            if(confirm("Supprimer cette affaire ?")) {
                state.deals = state.deals.filter(d => d.id !== id);
                updateAll();
            }
        }

        function editDeal(id) {
            const d = state.deals.find(x => x.id === id);
            document.getElementById('edit-id').value = d.id;
            document.getElementById('d-title').value = d.title;
            document.getElementById('d-amount').value = d.price;
            document.getElementById('d-date').value = d.date;
            openDealModal();
        }

        function saveDeal() {
            const id = document.getElementById('edit-id').value;
            const data = {
                id: id ? parseInt(id) : Date.now(),
                title: document.getElementById('d-title').value,
                user: document.getElementById('d-user').value,
                price: parseFloat(document.getElementById('d-amount').value),
                date: document.getElementById('d-date').value,
                stage: id ? state.deals.find(x=>x.id==id).stage : 'prospection'
            };

            if(id) {
                const idx = state.deals.findIndex(x => x.id == id);
                state.deals[idx] = data;
            } else {
                state.deals.push(data);
            }
            closeModals();
            updateAll();
        }

        function moveDeal(id, stage) {
            state.deals.find(d => d.id === id).stage = stage;
            updateAll();
        }

        // --- GESTION ADMIN / UTILISATEURS ---
        function renderAdminTable() {
            document.getElementById('user-table-body').innerHTML = state.users.map(u => `
                <tr>
                    <td>${u.name}</td><td>${u.role}</td><td>${u.rate}%</td>
                    <td><button onclick="deleteUser(${u.id})" class="btn btn-danger" style="padding:5px 10px;"><i class="fas fa-trash"></i></button></td>
                </tr>
            `).join('');
        }

        function saveUser() {
            state.users.push({
                id: Date.now(),
                name: document.getElementById('u-name').value,
                role: document.getElementById('u-role').value,
                rate: parseFloat(document.getElementById('u-rate').value) || 0
            });
            closeModals();
            updateAll();
        }

        function deleteUser(id) {
            state.users = state.users.filter(u => u.id !== id);
            updateAll();
        }

        // --- CHARTS & BI ---
        function renderBI() {
            const ctx = document.getElementById('mainChart').getContext('2d');
            if(currentChart) currentChart.destroy();

            const months = ["01","02","03","04","05","06","07","08","09","10","11","12"];
            const caValues = months.map(m => {
                return state.deals.filter(d => d.date.startsWith(`2026-${m}`) && d.stage === 'gagne').reduce((s,d)=>s+d.price, 0);
            });

            currentChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ["Jan", "Fév", "Mar", "Avr", "Mai", "Juin", "Juil", "Août", "Sep", "Oct", "Nov", "Déc"],
                    datasets: [{ label: 'CA Gagné 2026', data: caValues, backgroundColor: '#4f46e5' }]
                },
                options: { responsive: true, maintainAspectRatio: false }
            });
        }

        // --- HELPERS ---
        function openDealModal() {
            document.getElementById('d-user').innerHTML = state.users.map(u => `<option>${u.name}</option>`).join('');
            document.getElementById('dealModal').style.display = 'flex';
        }
        function openModal(id) { document.getElementById(id).style.display = 'flex'; }
        function closeModals() { 
            document.querySelectorAll('.modal').forEach(m => m.style.display = 'none');
            document.getElementById('edit-id').value = ""; 
        }
    </script>
